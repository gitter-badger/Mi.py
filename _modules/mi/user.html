[ドキュメント]    @staticmethod
    def follow(user_id: Optional[str]) -> tuple[bool, str]:
        """
        与えられたIDのユーザーをフォローします

        Parameters
        ----------
        user_id : Optional[str] = None
            フォローしたいユーザーのID

        Returns
        -------
        status: bool = False
            成功ならTrue, 失敗ならFalse
        """
        data = {"userId": user_id}
        res = api("/api/following/create", json_data=data, auth=True)
        if res.json().get("error"):
            code = res.json()["error"]["code"]
            status = False
        else:
            code = None
            status = True
        return status, code

[ドキュメント]    def get_profile(self) -> "UserProfile":
        """
        ユーザーのプロフィールを取得します

        Returns
        -------
        UserProfile:
            ユーザーのプロフィールオブジェクト

        return UserProfile(

                Controller.get_user(user_id=self.id, username=self.username,
